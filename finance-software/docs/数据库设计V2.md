## 1. 数据库架构

### 1.1 多账套设计

采用单一数据库，通过 tenant_id 字段隔离不同账套数据：

```sql
-- 公共表（租户无关）
- users (用户)
- roles (角色)
- tenants (账套)

-- 业务表（按 tenant_id 隔离）
- bank_accounts (账户)
- vouchers (凭证)
- voucher_items (凭证分录)
- invoices (发票)
- tax_records (税务记录)
```

### 1.2 年度隔离

每个业务表包含 year 字段，支持同一账套不同年度数据隔离：

```sql
ALTER TABLE bank_accounts ADD COLUMN year INT;
ALTER TABLE vouchers ADD COLUMN year INT;
ALTER TABLE voucher_items ADD COLUMN year INT;
```

## 2. 核心表结构

### 2.1 账套表 (tenants)

```sql
CREATE TABLE tenants (
  id VARCHAR(36) PRIMARY KEY,
  code VARCHAR(50) UNIQUE NOT NULL,     -- 账套编码
  name VARCHAR(200) NOT NULL,          -- 公司名称
  tax_no VARCHAR(50),                   -- 税号
  contact VARCHAR(100),                 -- 联系人
  phone VARCHAR(20),                    -- 电话
  address VARCHAR(500),                 -- 地址
  status VARCHAR(20) DEFAULT 'active', -- 状态: active/inactive
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### 2.2 会计期间表 (accounting_periods)

```sql
CREATE TABLE accounting_periods (
  id VARCHAR(36) PRIMARY KEY,
  tenant_id VARCHAR(36) NOT NULL,      -- 账套ID
  year INT NOT NULL,                   -- 年度
  month INT,                           -- 月份（1-12）
  period_name VARCHAR(50),             -- 期间名称
  start_date DATE NOT NULL,            -- 开始日期
  end_date DATE NOT NULL,              -- 结束日期
  status VARCHAR(20) DEFAULT 'open',  -- 状态: open/closed/year_closed
  is_adjustment BOOLEAN DEFAULT FALSE,-- 是否调整期
  created_at TIMESTAMP DEFAULT NOW(),
  FOREIGN KEY (tenant_id) REFERENCES tenants(id)
);
```

### 2.3 科目表 (accounts)

```sql
CREATE TABLE accounts (
  id VARCHAR(36) PRIMARY KEY,
  tenant_id VARCHAR(36) NOT NULL,
  year INT NOT NULL,
  code VARCHAR(20) NOT NULL,           -- 科目编码
  name VARCHAR(100) NOT NULL,          -- 科目名称
  level INT DEFAULT 1,                 -- 科目级次
  parent_code VARCHAR(20),             -- 上级科目编码
  direction VARCHAR(10) DEFAULT 'debit', -- 余额方向: debit/credit
  is_auxiliary BOOLEAN DEFAULT FALSE,  -- 是否辅助核算
  aux_type VARCHAR(50),               -- 辅助核算类型: dept/project/person
  is_cash BOOLEAN DEFAULT FALSE,      -- 是否现金流量科目
  is_quantity BOOLEAN DEFAULT FALSE,  -- 是否数量金额核算
  unit VARCHAR(20),                    -- 计量单位
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  FOREIGN KEY (tenant_id) REFERENCES tenants(id)
);
```

### 2.4 账户表 (bank_accounts)

```sql
CREATE TABLE bank_accounts (
  id VARCHAR(36) PRIMARY KEY,
  tenant_id VARCHAR(36),
  year INT,
  code VARCHAR(50) NOT NULL,
  name VARCHAR(100) NOT NULL,
  type VARCHAR(20) NOT NULL,          -- bank/cash/wechat/alipay
  balance DECIMAL(15,2) DEFAULT 0,
  currency VARCHAR(10) DEFAULT 'CNY',
  bank_name VARCHAR(200),
  account_no VARCHAR(50),
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### 2.5 凭证表 (vouchers)

```sql
CREATE TABLE vouchers (
  id VARCHAR(36) PRIMARY KEY,
  tenant_id VARCHAR(36),
  year INT,
  voucher_no VARCHAR(50) NOT NULL,   -- 凭证号
  voucher_date DATE NOT NULL,          -- 凭证日期
  status VARCHAR(20) DEFAULT 'draft', -- draft/approved/posted
  attach_count INT DEFAULT 0,         -- 附件张数
  total_debit DECIMAL(15,2),           -- 借方合计
  total_credit DECIMAL(15,2),         -- 贷方合计
  attachment VARCHAR(500),            -- 附件路径
  template_type VARCHAR(50),          -- 模板类型
  maker_id VARCHAR(36),
  maker_name VARCHAR(100),
  checker_id VARCHAR(36),
  checker_name VARCHAR(100),
  poster_id VARCHAR(36),
  poster_name VARCHAR(100),
  remark VARCHAR(500),
  period_id VARCHAR(36),              -- 会计期间ID
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### 2.6 凭证分录表 (voucher_items)

```sql
CREATE TABLE voucher_items (
  id VARCHAR(36) PRIMARY KEY,
  voucher_id VARCHAR(36) NOT NULL,
  tenant_id VARCHAR(36),
  year INT,
  line_no INT NOT NULL,               -- 行号
  account_code VARCHAR(20) NOT NULL,  -- 科目编码
  account_name VARCHAR(100) NOT NULL,  -- 科目名称
  direction VARCHAR(10) NOT NULL,      -- 借方/贷方: debit/credit
  amount DECIMAL(15,2) NOT NULL,      -- 金额
  description VARCHAR(200),           -- 摘要
  aux_info VARCHAR(200),              -- 辅助核算信息
  created_at TIMESTAMP DEFAULT NOW(),
  FOREIGN KEY (voucher_id) REFERENCES vouchers(id)
);
```

### 2.7 凭证模板表 (voucher_templates)

```sql
CREATE TABLE voucher_templates (
  id VARCHAR(36) PRIMARY KEY,
  code VARCHAR(50) NOT NULL,
  name VARCHAR(100) NOT NULL,
  category VARCHAR(50),               -- 模板分类
  description VARCHAR(500),
  items JSONB,                        -- 分录模板JSON
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### 2.8 发票表 (invoices)

```sql
CREATE TABLE invoices (
  id VARCHAR(36) PRIMARY KEY,
  tenant_id VARCHAR(36),
  year INT,
  invoice_no VARCHAR(50) NOT NULL,   -- 发票号码
  invoice_code VARCHAR(50),           -- 发票代码
  type VARCHAR(20) NOT NULL,          -- VAT_SPECIAL/VAT_NORMAL/RECEIPT
  amount DECIMAL(15,2) NOT NULL,      -- 含税金额
  tax_rate DECIMAL(5,4),              -- 税率
  tax_amount DECIMAL(15,2),           -- 税额
  excluding_tax_amount DECIMAL(15,2), -- 不含税金额
  seller_name VARCHAR(200),           -- 销售方
  seller_tax_no VARCHAR(50),          -- 销售方税号
  buyer_name VARCHAR(200),            -- 购买方
  buyer_tax_no VARCHAR(50),          -- 购买方税号
  date DATE,                          -- 开票日期
  status VARCHAR(20) DEFAULT 'pending', -- pending/used/verified
  attachment VARCHAR(500),           -- 附件
  voucher_id VARCHAR(36),            -- 关联凭证ID
  ocr_data JSONB,                     -- OCR识别数据
  remark VARCHAR(500),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### 2.9 税务记录表 (tax_records)

```sql
CREATE TABLE tax_records (
  id VARCHAR(36) PRIMARY KEY,
  tenant_id VARCHAR(36),
  year INT,
  month INT NOT NULL,
  tax_type VARCHAR(20) NOT NULL,      -- VAT/INCOME/OTHER
  sales_tax DECIMAL(15,2),             -- 销项税额
  purchase_tax DECIMAL(15,2),         -- 进项税额
  tax_amount DECIMAL(15,2),           -- 应纳税额
  tax_rate DECIMAL(5,4),              -- 税负率
  status VARCHAR(20) DEFAULT 'pending',
  remark VARCHAR(500),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### 2.10 用户表 (users)

```sql
CREATE TABLE users (
  id VARCHAR(36) PRIMARY KEY,
  username VARCHAR(50) UNIQUE NOT NULL,
  password VARCHAR(200) NOT NULL,
  real_name VARCHAR(100),
  email VARCHAR(100),
  phone VARCHAR(20),
  is_super_admin BOOLEAN DEFAULT FALSE,
  tenant_id VARCHAR(36),             -- 所属账套（普通用户）
  status VARCHAR(20) DEFAULT 'active',
  last_login_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### 2.11 角色表 (roles)

```sql
CREATE TABLE roles (
  id VARCHAR(36) PRIMARY KEY,
  code VARCHAR(50) UNIQUE NOT NULL,
  name VARCHAR(100) NOT NULL,
  description VARCHAR(500),
  is_system BOOLEAN DEFAULT FALSE,
  permissions JSONB,                  -- 权限列表
  created_at TIMESTAMP DEFAULT NOW()
);
```

### 2.12 用户角色关联表 (user_roles)

```sql
CREATE TABLE user_roles (
  id VARCHAR(36) PRIMARY KEY,
  user_id VARCHAR(36) NOT NULL,
  role_id VARCHAR(36) NOT NULL,
  created_at TIMESTAMP DEFAULT NOW()
);
```

## 4. 索引设计

```sql
-- 凭证相关索引
CREATE INDEX idx_vouchers_tenant_year ON vouchers(tenant_id, year);
CREATE INDEX idx_vouchers_date ON vouchers(voucher_date);
CREATE INDEX idx_vouchers_status ON vouchers(status);

-- 凭证分录索引
CREATE INDEX idx_voucher_items_voucher ON voucher_items(voucher_id);
CREATE INDEX idx_voucher_items_account ON voucher_items(account_code);

-- 发票索引
CREATE INDEX idx_invoices_tenant_year ON invoices(tenant_id, year);
CREATE INDEX idx_invoices_no ON invoices(invoice_no);

-- 科目索引
CREATE INDEX idx_accounts_tenant_year ON accounts(tenant_id, year);
CREATE INDEX idx_accounts_code ON accounts(code);
```
